同心圆

import turtle

n=int(input("输入圆的个数:"))

bangjing=25

jianju=10#一个常量

sudu=0

turtle.speed(sudu)#绘制速度

turtle.hideturtle()#隐藏光标

a=bangjing

for count in range(n):

turtle.circle(a)#半径

x=turtle.xcor()#目前光标的x位置

y=turtle.ycor()-jianju#调整目前光标的位置

a=a+jianju#调正半径

turtle.penup()#提笔

turtle.goto(x,y)#调整画笔位置

turtle.pendown()#落笔

turtle.done()
